    let sql = "SELECT NOMBRE_ADMIN FROM admins WHERE CORREO = '"+req.body.CORREO+"' AND PASSWORD_ADMIN = SHA1('"+req.body.PASSWORD_ADMIN+"') AND ESTADO_ADMIN = '1' ";
    conexion.query(sql, function (error, results) {
        if(error){
            throw error;
        }else{
            if(Object.keys(results).length === 0 ){
                res.status(201).json({ error: '201' });
                
            }else{
                res.status(200).json(results);
            }
            
            
        }
    })
